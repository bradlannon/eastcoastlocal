2. In src/app/api/events/route.ts: Import `event_sources` from schema and `sql` from drizzle-orm. Modify the query to left-join event_sources and aggregate source_types per event.

The cleanest approach: Keep the existing query, then do a single supplementary query to get source_types for all returned event IDs:

```typescript
import { events, venues, event_sources } from '@/lib/db/schema';
import { gte, eq, inArray, sql } from 'drizzle-orm';

// Existing query
const rows = await db
  .select()
  .from(events)
  .innerJoin(venues, eq(events.venue_id, venues.id))
  .where(gte(events.event_date, new Date()))
  .orderBy(events.event_date);

// Supplementary: get source_types for these events
const eventIds = rows.map(r => r.events.id);
const sourceRows = eventIds.length > 0
  ? await db
      .select({
        event_id: event_sources.event_id,
        source_type: event_sources.source_type,
      })
      .from(event_sources)
      .where(inArray(event_sources.event_id, eventIds))
  : [];

// Build lookup map
const sourceMap = new Map<number, string[]>();
for (const sr of sourceRows) {
  const arr = sourceMap.get(sr.event_id) ?? [];
  arr.push(sr.source_type);
  sourceMap.set(sr.event_id, arr);
}

// Merge into response
const enriched = rows.map(r => ({
  ...r,
  source_types: sourceMap.get(r.events.id) ?? [],
}));

return NextResponse.json(enriched);
```

This approach avoids duplicating event rows (which a LEFT JOIN + GROUP BY would cause with Drizzle's select-all pattern).
