export async function approveCandidate(formData: FormData): Promise<void> {
  const raw = formData.get('id');
  const id = parseInt(String(raw ?? ''), 10);
  if (isNaN(id)) return;
  try {
    await promoteSource(id);
  } catch (err) {
    console.error('[approveCandidate] promoteSource failed:', err);
  }
  revalidatePath('/admin/discovery');
  redirect('/admin/discovery');
}

**Part A: Server action (actions.ts)**

Add `batchApproveCandidate` server action to actions.ts:
- Accept FormData with an 'ids' field containing comma-separated numeric IDs
- Parse: `String(raw).split(',').map(Number).filter(n => !isNaN(n))`
- If empty array, return early
- Use `Promise.allSettled(ids.map(id => promoteSource(id)))` — NOT Promise.all (per research: allSettled continues on individual failures)
- Log failures count if any: `console.error('[batchApproveCandidate] ${failures.length} of ${ids.length} failed')`
- Call `revalidatePath('/admin/discovery')` then `redirect('/admin/discovery')` AFTER the try/catch (redirect throws internally — must not be inside try/catch per existing pattern and Pitfall 6)
